


































































1 CentroCostosOPM.java (Entity)
Representa la entidad de negocio y el mapeo de la tabla CRE_CAT_CENTRO_COSTOS_OPM.
Cada atributo Java corresponde a una columna de la tabla. 


2 CentroCostosOPMDAO.java (DAO / Persistencia)
Define las operaciones de acceso a datos hacia la base de datos.
Ejemplos:
guardar, 
actualizar, 
eliminar, 
consultar. 
Aquí NO va lógica de negocio, solo persistencia.

3 CentroCostosOPMBI.java (Business Interface)
Define los métodos de lógica de negocio que el sistema puede ejecutar.
Ejemplos:
alta de centro de costos, 
consulta general, 
actualización, 
eliminación lógica. 
Es el contrato de negocio.

4 CentroCostosOPMBIImpl.java (Business Implementation)
Implementa la lógica de negocio real y las validaciones del sistema.
Ejemplos:
validar datos obligatorios, 
activar/desactivar registros, 
decidir reglas del negocio, 
llamar al DAO para persistencia. 
Ejemplo:
"La clave de centro de costos es obligatoria".


5 CentroCostosOPMAppl.java (Application Interface)
Define la comunicación entre frontend y lógica de negocio.
Funciona como capa intermedia entre presentación y negocio. 


6 CentroCostosOPMApplImpl.java (Application Implementation)
Implementa la capa aplicación y delega las operaciones hacia la capa de negocio (BI).
Su función principal es:
recibir solicitudes del frontend, 
enviar la petición a negocio, 
regresar la respuesta. 
No contiene lógica pesada; actúa como coordinador/intermediario.


7 CentroCostosOPM.jsp (Vista / Presentación)
Página JSP que carga la interfaz frontend.
Se encarga de: 
cargar scripts, 
renderizar componentes, 
preparar el contenedor visual. 


8 CentroCostosOPM.js (Frontend / ExtJS)
Contiene la lógica visual y comportamiento frontend.
Maneja: 
botones CRUD, 
formularios, 
grids, 
eventos, 
llamadas DWR al backend, 
carga de datos en pantalla.

# 4. `CentroCostosOPMBIImpl.java`

```java
package com.inbursa.creditos.business.impl;

import java.util.List;

import com.inbursa.creditos.business.CentroCostosOPMBI;
import com.inbursa.creditos.persistence.dao.CentroCostosOPMDAO;
import com.inbursa.creditos.persistence.entity.CentroCostosOPM;

/*
 * Implementación de la lógica de negocio
 * para el módulo CentroCostosOPM.
 */
public class CentroCostosOPMBIImpl implements CentroCostosOPMBI {

    /*
     * DAO utilizado para operaciones de persistencia.
     */
    private CentroCostosOPMDAO centroCostosOPMDAO;

    /*
     * Realiza el alta de un centro de costos.
     * Valida información y activa el registro.
     */
    public void altaCentroCostos(CentroCostosOPM dto) throws Exception {
        validar(dto);
        dto.setBolActivo(true);
        centroCostosOPMDAO.guardar(dto);
    }

    /*
     * Actualiza la información del registro.
     */
    public void actualizarCentroCostos(CentroCostosOPM dto) throws Exception {
        validar(dto);
        centroCostosOPMDAO.actualizar(dto);
    }

    /*
     * Ejecuta eliminación lógica cambiando el estatus a inactivo.
     */
    public void eliminarCentroCostos(String clave) throws Exception {
        CentroCostosOPM entity = centroCostosOPMDAO.obtenerPorClave(clave);
        if(entity != null){
            entity.setBolActivo(false);
            centroCostosOPMDAO.actualizar(entity);
        }
    }

    /*
     * Consulta un registro por clave.
     */
    public CentroCostosOPM consultarPorClave(String clave) throws Exception {
        return centroCostosOPMDAO.obtenerPorClave(clave);
    }

    /*
     * Obtiene todos los registros disponibles.
     */
    public List<CentroCostosOPM> consultarTodos() throws Exception {
        return centroCostosOPMDAO.consultarTodos();
    }

    /*
     * Consulta registros utilizando filtros.
     */
    public List<CentroCostosOPM> consultarPorFiltro(String codigo, Boolean activo) throws Exception {
        return centroCostosOPMDAO.consultarPorFiltro(codigo, activo);
    }

    /*
     * Método central de validaciones de negocio.
     */
    private void validar(CentroCostosOPM dto) throws Exception {

        /*
         * Valida que el objeto exista.
         */
        if(dto == null){
            throw new Exception("El objeto CentroCostosOPM es requerido");
        }

        /*
         * Valida que la clave sea obligatoria.
         */
        if(dto.getCveCentroCostos() == null || dto.getCveCentroCostos().trim().equals("")){
            throw new Exception("La clave de centro de costos es obligatoria");
        }
    }

    /*
     * Inyección del DAO.
     */
    public void setCentroCostosOPMDAO(CentroCostosOPMDAO centroCostosOPMDAO) {
        this.centroCostosOPMDAO = centroCostosOPMDAO;
    }
}
```

# 8. `CentroCostosOPM.js`

```javascript
/*
 * Inicialización principal del módulo una vez cargado el DOM.
 */
Ext.onReady(function(){

    /*
     * Store utilizado para almacenar registros del grid.
     */
    var storeCentroCostos = new Ext.data.JsonStore({

        /*
         * Definición de campos del modelo.
         */
        fields:[
            'cveCentroCostos',
            'numOrden',
            'numCuentaMaxima',
            'cuentaOperativa',
            'bolActivo',
            'codigo'
        ],

        /*
         * Inicialización vacía del store.
         */
        data:[]
    });

    /*
     * Función encargada de consultar registros del backend.
     */
    function cargarDatos(){

        /*
         * Llamada DWR para obtener registros.
         */
        CentroCostosOPMDWR.obtenerTodos(function(response){

            /*
             * Carga de información en store.
             */
            storeCentroCostos.loadData(response);
        });
    }

    /*
     * Formulario principal de captura.
     */
    var formCentroCostos = new Ext.form.FormPanel({

        title:'Centro Costos OPM',

        /*
         * Campos del formulario.
         */
        items:[
            {xtype:'textfield', fieldLabel:'Clave', name:'cveCentroCostos'},
            {xtype:'textfield', fieldLabel:'Código', name:'codigo'},
            {xtype:'textfield', fieldLabel:'Cuenta Operativa', name:'cuentaOperativa'}
        ],

        /*
         * Botones de operación CRUD.
         */
        buttons:[
            {
                text:'Guardar',

                /*
                 * Evento guardar registro.
                 */
                handler:function(){

                    var values = formCentroCostos.getForm().getValues();

                    CentroCostosOPMDWR.guardarCentroCostos(values, function(){

                        cargarDatos();
                    });
                }
            },
            {
                text:'Actualizar',

                /*
                 * Evento actualizar registro.
                 */
                handler:function(){

                    var values = formCentroCostos.getForm().getValues();

                    CentroCostosOPMDWR.actualizarCentroCostos(values, function(){

                        cargarDatos();
                    });
                }
            }
        ]
    });

    /*
     * Grid principal de visualización.
     */
    var gridCentroCostos = new Ext.grid.GridPanel({

        /*
         * Renderizado sobre contenedor JSP.
         */
        renderTo:'gridCentroCostos',

        /*
         * Fuente de datos del grid.
         */
        store:storeCentroCostos,

        /*
         * Definición de columnas.
         */
        columns:[
            {header:'Clave', dataIndex:'cveCentroCostos'},
            {header:'Código', dataIndex:'codigo'},
            {header:'Cuenta Operativa', dataIndex:'cuentaOperativa'},
            {header:'Activo', dataIndex:'bolActivo'}
        ],

        height:300,
        width:800,

        /*
         * Barra superior de acciones.
         */
        tbar:[
            {
                text:'Eliminar',

                /*
                 * Evento eliminación lógica.
                 */
                handler:function(){

                    var record = gridCentroCostos.getSelectionModel().getSelected();

                    if(record){

                        CentroCostosOPMDWR.eliminarCentroCostos(record.data.cveCentroCostos, function(){

                            cargarDatos();
                        });
                    }
                }
            }
        ]
    });

    /*
     * Renderizado del formulario.
     */
    formCentroCostos.render(document.body);

    /*
     * Carga inicial de información.
     */
    cargarDatos();
});
```
