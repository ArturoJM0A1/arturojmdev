## 4. `CentroCostosOPMBIImpl.java`

```java
package com.inbursa.creditos.business.impl;

import java.util.List;

import com.inbursa.creditos.business.CentroCostosOPMBI;
import com.inbursa.creditos.persistence.dao.CentroCostosOPMDAO;
import com.inbursa.creditos.persistence.entity.CentroCostosOPM;

public class CentroCostosOPMBIImpl implements CentroCostosOPMBI {

    private CentroCostosOPMDAO centroCostosOPMDAO;

    public void altaCentroCostos(CentroCostosOPM dto) throws Exception {
        validar(dto);
        dto.setBolActivo(true);
        centroCostosOPMDAO.guardar(dto);
    }

    public void actualizarCentroCostos(CentroCostosOPM dto) throws Exception {
        validar(dto);
        centroCostosOPMDAO.actualizar(dto);
    }

    public void eliminarCentroCostos(String clave) throws Exception {
        CentroCostosOPM entity = centroCostosOPMDAO.obtenerPorClave(clave);
        if(entity != null){
            entity.setBolActivo(false);
            centroCostosOPMDAO.actualizar(entity);
        }
    }

    public CentroCostosOPM consultarPorClave(String clave) throws Exception {
        return centroCostosOPMDAO.obtenerPorClave(clave);
    }

    public List<CentroCostosOPM> consultarTodos() throws Exception {
        return centroCostosOPMDAO.consultarTodos();
    }

    public List<CentroCostosOPM> consultarPorFiltro(String codigo, Boolean activo) throws Exception {
        return centroCostosOPMDAO.consultarPorFiltro(codigo, activo);
    }

    private void validar(CentroCostosOPM dto) throws Exception {
        if(dto == null){
            throw new Exception("El objeto CentroCostosOPM es requerido");
        }
        if(dto.getCveCentroCostos() == null || dto.getCveCentroCostos().trim().equals("")){
            throw new Exception("La clave de centro de costos es obligatoria");
        }
    }

    public void setCentroCostosOPMDAO(CentroCostosOPMDAO centroCostosOPMDAO) {
        this.centroCostosOPMDAO = centroCostosOPMDAO;
    }
}
```

## 8. `CentroCostosOPM.js`

```javascript
Ext.onReady(function(){

    var storeCentroCostos = new Ext.data.JsonStore({
        fields:[
            'cveCentroCostos',
            'numOrden',
            'numCuentaMaxima',
            'cuentaOperativa',
            'bolActivo',
            'codigo'
        ],
        data:[]
    });

    function cargarDatos(){
        CentroCostosOPMDWR.obtenerTodos(function(response){
            storeCentroCostos.loadData(response);
        });
    }

    var formCentroCostos = new Ext.form.FormPanel({
        title:'Centro Costos OPM',
        items:[
            {xtype:'textfield', fieldLabel:'Clave', name:'cveCentroCostos'},
            {xtype:'textfield', fieldLabel:'Código', name:'codigo'},
            {xtype:'textfield', fieldLabel:'Cuenta Operativa', name:'cuentaOperativa'}
        ],
        buttons:[
            {
                text:'Guardar',
                handler:function(){
                    var values = formCentroCostos.getForm().getValues();
                    CentroCostosOPMDWR.guardarCentroCostos(values, function(){
                        cargarDatos();
                    });
                }
            },
            {
                text:'Actualizar',
                handler:function(){
                    var values = formCentroCostos.getForm().getValues();
                    CentroCostosOPMDWR.actualizarCentroCostos(values, function(){
                        cargarDatos();
                    });
                }
            }
        ]
    });

    var gridCentroCostos = new Ext.grid.GridPanel({
        renderTo:'gridCentroCostos',
        store:storeCentroCostos,
        columns:[
            {header:'Clave', dataIndex:'cveCentroCostos'},
            {header:'Código', dataIndex:'codigo'},
            {header:'Cuenta Operativa', dataIndex:'cuentaOperativa'},
            {header:'Activo', dataIndex:'bolActivo'}
        ],
        height:300,
        width:800,
        tbar:[
            {
                text:'Eliminar',
                handler:function(){
                    var record = gridCentroCostos.getSelectionModel().getSelected();
                    if(record){
                        CentroCostosOPMDWR.eliminarCentroCostos(record.data.cveCentroCostos, function(){
                            cargarDatos();
                        });
                    }
                }
            }
        ]
    });

    formCentroCostos.render(document.body);
    cargarDatos();
});
```
